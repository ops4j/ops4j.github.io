462     /**
463      * This method allows to create a {@link PaxUserAdmin} service for
464      * testpurpose
465      * 
466      * @return
467      */
468     @SuppressWarnings({ "rawtypes" })
469     private static PaxUserAdmin createUserAdmin(StorageProvider storageProvider, LogService logService, EventAdmin eventAdmin) {
470         ServiceTracker<UserAdminListener, UserAdminListener> listenerTracker = EasyMock.createNiceMock(ServiceTracker.class);
471         EasyMock.expect(listenerTracker.getService()).andStubReturn(null);
472         EasyMock.expect(listenerTracker.getService(EasyMock.<ServiceReference<UserAdminListener>> anyObject())).andStubReturn(null);
473         EasyMock.expect(listenerTracker.getServices()).andStubReturn(null);
474         EasyMock.expect(listenerTracker.getServices(EasyMock.<UserAdminListener[]> anyObject())).andStubReturn(new UserAdminListener[0]);
475         ServiceTracker<LogService, LogService> logServiceTracker = createTracker(logService);
476         //Create a reference to the Eventadmin
477         ServiceTracker<EventAdmin, EventAdmin> eventAdminTracker = createTracker(eventAdmin);
478         PaxUserAdmin userAdmin = new PaxUserAdmin(storageProvider, logServiceTracker, eventAdminTracker, listenerTracker, Executors.newCachedThreadPool());
479         //register it!
480         BundleContext bundleContext = EasyMock.createNiceMock(BundleContext.class);
481         ServiceRegistration serviceRegistration = EasyMock.createNiceMock(ServiceRegistration.class);
482         ServiceReference serviceReference = EasyMock.createNiceMock(ServiceReference.class);
483         EasyMock.expect(serviceRegistration.getReference()).andReturn(serviceReference);
484         EasyMock.expect(serviceReference.getProperty(Constants.SERVICE_ID)).andReturn("1");
485         EasyMock.expect(serviceReference.getProperty(Constants.OBJECTCLASS)).andReturn(UserAdmin.class.getName());
486         EasyMock.expect(serviceReference.getProperty(Constants.SERVICE_PID)).andReturn(PaxUserAdminConstants.SERVICE_PID);
487         EasyMock.expect(bundleContext.registerService(EasyMock.eq(UserAdmin.class), EasyMock.eq(userAdmin), EasyMock.<Dictionary<String, Object>> anyObject())).andReturn(serviceRegistration);
488         EasyMock.replay(listenerTracker, logServiceTracker, eventAdminTracker, bundleContext, serviceRegistration, serviceReference);
489         userAdmin.register(bundleContext, "junit", 0l);
490         EasyMock.verify(bundleContext);
491         EasyMock.reset(bundleContext);
492         return userAdmin;
493     }
