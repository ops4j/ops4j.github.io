20  import java.util.Dictionary;

23  import org.ops4j.pax.useradmin.provider.ldap.ConfigurationConstants;
24  import org.ops4j.pax.useradmin.service.spi.StorageProvider;
25  import org.osgi.framework.BundleActivator;
26  import org.osgi.framework.BundleContext;
27  
28  import com.novell.ldap.LDAPConnection;

30  /**
31   * Activator of the Pax UserAdmin LDAP StorageProvider bundle.
32   * 
33   * @author Matthias Kuespert
34   * @since 08.07.2009
35   */
36  public class Activator implements BundleActivator {
37  
38      /**
39       * Create and register the <code>StorageProvider</code> service.
40       * 
41       * @see org.osgi.framework.BundleActivator#start(org.osgi.framework.BundleContext)
42       */
43      @Override
44      public void start(BundleContext context) throws Exception {
45          //
46          // create LDAP connection access object
47          //
48          LDAPConnection connection = new LDAPConnection();
49          //
50          // set service properties
51          //
52          Dictionary<String, String> properties = new Hashtable<String, String>();
53          properties.put(org.ops4j.pax.useradmin.service.PaxUserAdminConstants.STORAGEPROVIDER_TYPE, ConfigurationConstants.STORAGEPROVIDER_TYPE);
54          //
55          // create & register service implementation
56          //
57          context.registerService(StorageProvider.class, new StorageProviderImpl(connection), properties);
58      }
59  
60      /**
61       * @see org.osgi.framework.BundleActivator#stop(org.osgi.framework.BundleContext)
62       */
63      @Override
64      public void stop(BundleContext context) throws Exception {
65      }
